#include <M5Cardputer.h>

// --- 常量定义 ---
const char UP_KEY_STR = '.';
const char DOWN_KEY_STR = ';';

// --- 颜色定义 ---
const uint16_t COLOR_DEFAULT = TFT_WHITE;  // 白色
const uint16_t COLOR_FOCUSED = TFT_GREEN;  // 绿色（选中）
const uint16_t COLOR_BACKGROUND = TFT_BLACK;  // 黑色
const uint16_t COLOR_TITLE_BG = TFT_BLUE;  // 标题背景色

// --- 全局变量 ---
// 状态变量
char current_mode = 'i';
char priority_mode = 'A';
float last_lux_value = 0;

// 预定义值
const int iso_values[] = {50, 100, 200, 400, 800, 1000, 1200, 1600, 3200, 6400};
const int iso_count = sizeof(iso_values) / sizeof(iso_values[0]);

const float aperture_values[] = {0.95, 1.0, 1.2, 1.4, 1.8, 2.0, 2.2, 2.8, 4, 5.6, 8, 11, 16, 22};
const int aperture_count = sizeof(aperture_values) / sizeof(aperture_values[0]);

const char* shutter_values[] = {"30s", "15s", "8s", "4s", "2s", "1s", "1/2", "1/4", "1/8", "1/15", "1/30", "1/60", "1/125", "1/250", "1/500", "1/1000", "1/2000", "1/4000"};
const int shutter_count = sizeof(shutter_values) / sizeof(shutter_values[0]);

// 当前预定义值的索引
int preview_indices[3] = {1, 4, 11};  // ISO, Aperture, Shutter

void draw_ui() {
  // 清空屏幕
  M5Cardputer.Display.fillScreen(COLOR_BACKGROUND);
  
  // 绘制标题栏
  M5Cardputer.Display.fillRect(0, 0, 240, 22, COLOR_TITLE_BG);
  M5Cardputer.Display.setTextColor(COLOR_DEFAULT, COLOR_TITLE_BG);
  M5Cardputer.Display.setCursor(10, 5);
  M5Cardputer.Display.print("LightMeter");
  
  // 绘制静态标签
  M5Cardputer.Display.setTextColor(COLOR_DEFAULT, COLOR_BACKGROUND);
  M5Cardputer.Display.setCursor(8, 28);
  M5Cardputer.Display.print("LUX:");
  M5Cardputer.Display.setCursor(8, 54);
  M5Cardputer.Display.print("ISO:");
  M5Cardputer.Display.setCursor(8, 82);
  M5Cardputer.Display.print("APER:");
  M5Cardputer.Display.setCursor(8, 110);
  M5Cardputer.Display.print("SSPD:");
  
  // 绘制当前模式
  M5Cardputer.Display.setTextColor(COLOR_FOCUSED, COLOR_BACKGROUND);
  if (current_mode == 'i') {
    M5Cardputer.Display.setCursor(8, 54);
    M5Cardputer.Display.print("ISO:");
  } else if (current_mode == 'a') {
    M5Cardputer.Display.setCursor(8, 82);
    M5Cardputer.Display.print("APER:");
  } else if (current_mode == 's') {
    M5Cardputer.Display.setCursor(8, 110);
    M5Cardputer.Display.print("SSPD:");
  }
  
  // 显示当前值
  M5Cardputer.Display.setTextColor(COLOR_DEFAULT, COLOR_BACKGROUND);
  M5Cardputer.Display.setCursor(82, 54);
  M5Cardputer.Display.print(iso_values[preview_indices[0]]);
  
  M5Cardputer.Display.setCursor(82, 82);
  M5Cardputer.Display.print("f/");
  M5Cardputer.Display.print(aperture_values[preview_indices[1]], 1);
  
  M5Cardputer.Display.setCursor(82, 110);
  M5Cardputer.Display.print(shutter_values[preview_indices[2]]);
  
  // 显示电池电量
  M5Cardputer.Display.setTextColor(COLOR_DEFAULT, COLOR_TITLE_BG);
  M5Cardputer.Display.setCursor(188, 5);
  M5Cardputer.Display.print("B:");
  M5Cardputer.Display.setCursor(200, 5);
  M5Cardputer.Display.print(String(M5Cardputer.Power.getBatteryLevel()) + "%");
}

void setup() {
  M5Cardputer.begin();
  M5Cardputer.Display.setBrightness(200);
  M5Cardputer.Display.setTextSize(1);
  M5Cardputer.Display.setTextColor(TFT_WHITE, TFT_BLACK);
  
  // 初始化UI
  draw_ui();
}

void loop() {
  M5Cardputer.update();
  
  // 键盘事件处理
  if (M5Cardputer.Keyboard.isChange()) {
    // 检查模式切换键
    if (M5Cardputer.Keyboard.isKeyPressed('i')) {
      current_mode = 'i';
      draw_ui();
    } else if (M5Cardputer.Keyboard.isKeyPressed('a')) {
      current_mode = 'a';
      priority_mode = 'A';
      draw_ui();
    } else if (M5Cardputer.Keyboard.isKeyPressed('s')) {
      current_mode = 's';
      priority_mode = 'S';
      draw_ui();
    } else if (M5Cardputer.Keyboard.isKeyPressed('\t')) {
      // Tab键在i, a, s三个模式之间循环切换
      if (current_mode == 'i') {
        current_mode = 'a';
        priority_mode = 'A';
      } else if (current_mode == 'a') {
        current_mode = 's';
        priority_mode = 'S';
      } else if (current_mode == 's') {
        current_mode = 'i';
      }
      draw_ui();
    }
    
    // 检查滚动键（只简单更新索引，不实现动画）
    int param_index = current_mode == 'i' ? 0 : (current_mode == 'a' ? 1 : 2);
    int max_idx = current_mode == 'i' ? iso_count - 1 : (current_mode == 'a' ? aperture_count - 1 : shutter_count - 1);
    
    if (M5Cardputer.Keyboard.isKeyPressed('.')) {
      if (preview_indices[param_index] > 0) {
        preview_indices[param_index]--;
        draw_ui();
      }
    } else if (M5Cardputer.Keyboard.isKeyPressed(';')) {
      if (preview_indices[param_index] < max_idx) {
        preview_indices[param_index]++;
        draw_ui();
      }
    }
  }
  
  delay(20);
}
